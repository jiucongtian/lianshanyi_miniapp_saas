# Logger 系统改造示例

本文档提供项目中实际文件的改造示例，展示如何将 console.log 迁移到统一的 logger 系统。

## 改造步骤

### 步骤 1: 引入 logger

在文件顶部添加：
```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('模块名');
```

### 步骤 2: 替换所有 console 调用

将所有 `console.log/warn/error` 替换为对应的 `log.info/warn/error`

### 步骤 3: 优化日志结构

使用结构化数据代替字符串拼接，移除手动的日志前缀。

## 实际文件改造示例

### 示例 1: utils/userManager.js

**改造前：**
```javascript
const userManager = {
  async getUserInfo() {
    console.log('[UserManager] 获取用户信息');
    try {
      const result = await wx.cloud.callFunction({...});
      console.log('[UserManager] 用户信息:', result);
      return result;
    } catch (error) {
      console.error('[UserManager] 获取失败:', error);
      throw error;
    }
  },
  
  updateCache(userInfo) {
    console.log('[UserManager] 更新缓存:', userInfo.nickName);
    // ...
  }
};
```

**改造后：**
```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('UserManager');

const userManager = {
  async getUserInfo() {
    log.info('获取用户信息');
    
    log.timeStart('getUserInfo');
    try {
      const result = await wx.cloud.callFunction({...});
      log.timeEnd('getUserInfo');
      log.info('用户信息获取成功', { userId: result._id });
      return result;
    } catch (error) {
      log.error('获取用户信息失败', { error: error.message });
      throw error;
    }
  },
  
  updateCache(userInfo) {
    log.debug('更新用户缓存', { 
      userId: userInfo._id, 
      nickName: userInfo.nickName 
    });
    // ...
  }
};
```

**改造要点：**
- ✅ 移除了重复的 `[UserManager]` 前缀
- ✅ 使用结构化数据代替字符串拼接
- ✅ 添加了性能追踪
- ✅ 区分 info/debug 级别

### 示例 2: beans/UserBean.js

**改造前：**
```javascript
class UserBean {
  constructor(data) {
    if (!data._id) {
      console.warn('[UserBean] 缺少_id字段');
    }
    
    if (typeof data.profileQuota !== 'number') {
      console.error('[UserBean] profileQuota字段类型错误:', typeof data.profileQuota);
    }
    
    this._id = data._id || '';
    // ...
  }
  
  canCreateMore() {
    console.log('[UserBean] 检查配额:', this.usedProfiles, '/', this.profileQuota);
    return this.profileQuota === -1 || this.usedProfiles < this.profileQuota;
  }
}
```

**改造后：**
```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('UserBean');

class UserBean {
  constructor(data) {
    if (!data._id) {
      log.warn('缺少_id字段', { data });
    }
    
    if (typeof data.profileQuota !== 'number') {
      log.error('profileQuota字段类型错误', { 
        expected: 'number',
        actual: typeof data.profileQuota,
        value: data.profileQuota
      });
    }
    
    this._id = data._id || '';
    // ...
  }
  
  canCreateMore() {
    const canCreate = this.profileQuota === -1 || this.usedProfiles < this.profileQuota;
    log.debug('检查用户配额', {
      usedProfiles: this.usedProfiles,
      profileQuota: this.profileQuota,
      canCreate
    });
    return canCreate;
  }
}
```

**改造要点：**
- ✅ Bean 类使用 `createModuleLogger`（不继承 BaseClass）
- ✅ 详细的错误信息结构化
- ✅ 使用 debug 级别记录内部状态检查

### 示例 3: utils/eventBus.js

**改造前：**
```javascript
class EventBus {
  constructor() {
    this.events = {};
  }
  
  on(eventName, callback) {
    console.log(`[EventBus] 注册事件: ${eventName}`);
    // ...
  }
  
  emit(eventName, data) {
    console.log(`[EventBus] 触发事件: ${eventName}`, data);
    // ...
  }
  
  off(eventName, callback) {
    console.log(`[EventBus] 移除事件: ${eventName}`);
    // ...
  }
}
```

**改造后：**
```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('EventBus');

class EventBus {
  constructor() {
    this.events = {};
    log.info('EventBus 初始化');
  }
  
  on(eventName, callback) {
    log.debug('注册事件监听', { eventName });
    // ...
  }
  
  emit(eventName, data) {
    const listenerCount = (this.events[eventName] || []).length;
    log.debug('触发事件', { 
      eventName, 
      listenerCount,
      hasData: !!data 
    });
    // ...
  }
  
  off(eventName, callback) {
    log.debug('移除事件监听', { eventName });
    // ...
  }
}
```

**改造要点：**
- ✅ 事件系统使用 debug 级别（避免生产环境日志过多）
- ✅ 记录监听器数量等有用信息
- ✅ 避免记录完整 data（可能很大）

### 示例 4: utils/imageCacheManager.js

**改造前：**
```javascript
class ImageCacheManager {
  async cacheImage(cloudPath) {
    console.log('[ImageCache] 开始缓存图片:', cloudPath);
    
    try {
      const tempFilePath = await this._downloadFromCloud(cloudPath);
      console.log('[ImageCache] 下载完成:', tempFilePath);
      
      const savedPath = await this._saveToLocal(tempFilePath);
      console.log('[ImageCache] 保存成功:', savedPath);
      
      return savedPath;
    } catch (error) {
      console.error('[ImageCache] 缓存失败:', error);
      throw error;
    }
  }
  
  clearCache() {
    console.log('[ImageCache] 清理缓存');
    // ...
  }
}
```

**改造后：**
```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('ImageCache');

class ImageCacheManager {
  async cacheImage(cloudPath) {
    log.info('开始缓存图片', { cloudPath });
    
    log.timeStart(`cache-${cloudPath}`);
    try {
      const tempFilePath = await this._downloadFromCloud(cloudPath);
      log.debug('下载完成', { cloudPath, tempFilePath });
      
      const savedPath = await this._saveToLocal(tempFilePath);
      log.timeEnd(`cache-${cloudPath}`);
      log.info('图片缓存成功', { cloudPath, savedPath });
      
      return savedPath;
    } catch (error) {
      log.error('图片缓存失败', { 
        cloudPath, 
        error: error.message,
        errCode: error.errCode 
      });
      throw error;
    }
  }
  
  clearCache() {
    log.info('清理图片缓存');
    log.timeStart('clearCache');
    // ...
    log.timeEnd('clearCache');
  }
}
```

**改造要点：**
- ✅ 添加性能追踪（缓存操作通常需要关注性能）
- ✅ 记录错误码等详细信息
- ✅ 区分 info（关键流程）和 debug（中间步骤）

### 示例 5: pages/profile/index.js

```javascript
const { ProfileController } = require('../../controllers/ProfileController');

  },

    this.controller.initialize();
  },
  
  onPullDownRefresh() {
    console.log('[ProfilePage] 下拉刷新');

      wx.stopPullDownRefresh();
    });
  },
  
  onProfileTap(e) {

    console.log('[ProfilePage] 点击档案:', profileId);
    this.controller.selectProfile(profileId);
  }
});
```

**改造后：**
```javascript
const { ProfileController } = require('../../controllers/ProfileController');
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('ProfilePage');

Page({
  data: {
    profileList: []
  },
  
  onLoad(options) {
    log.info('页面加载', { options });
    this.controller = new ProfileController(this);
    this.controller.initialize();
  },
  
  onPullDownRefresh() {
    log.info('下拉刷新');
    this.controller.loadProfiles().then(() => {
      wx.stopPullDownRefresh();
      log.info('刷新完成');
    });
  },
  
  onProfileTap(e) {
    const profileId = e.currentTarget.dataset.id;
    log.info('点击档案', { profileId });
    this.controller.selectProfile(profileId);
  }
});
```

**改造要点：**
- ✅ 页面文件使用 `createModuleLogger`
- ✅ 记录用户交互行为（有助于分析用户路径）
- ✅ 保持简洁（主要逻辑在 Controller 中有日志）

```javascript
  validatePhone
};
```

**改造后：**
```javascript

const log = createModuleLogger('Util');

function formatTime(date) {
  log.debug('格式化时间', { date });
  // ...
}

function validatePhone(phone) {
  const isValid = /^1[3-9]\d{9}$/.test(phone);
  log.debug('验证手机号', { 
    phone: phone.substring(0, 3) + '****' + phone.substring(7), // 脱敏

  });
  return isValid;
}

module.exports = {
  formatTime,
  validatePhone
};
```

**改造要点：**
- ✅ 工具函数使用 debug 级别
- ✅ 敏感信息（手机号）手动脱敏
- ✅ 一个文件一个 log 实例，所有函数共享

## 批量改造技巧

### 技巧 1: 使用正则批量替换

在 VS Code 中使用正则替换：

**查找：** `console\.log\('\[(\w+)\] (.+?)',`
**替换为：** `log.info('$2',`

### 技巧 2: 确定模块名

模块名通常与文件名对应：
- `userManager.js` → `UserManager`
- `imageCacheManager.js` → `ImageCache`
- `baziCalculator.js` → `BaziCalculator`

### 技巧 3: 日志级别选择

| 原日志 | 新级别 | 说明 |
|--------|--------|------|
| `console.log('[模块] 调试信息')` | `log.debug()` | 调试信息、中间状态 |
| `console.log('[模块] 操作成功')` | `log.info()` | 关键流程、成功操作 |
| `console.warn('[模块] 警告')` | `log.warn()` | 警告信息 |
| `console.error('[模块] 错误')` | `log.error()` | 错误信息 |

### 技巧 4: 数据结构化

**字符串拼接 → 结构化数据**
```javascript
// 改造前
console.log(`用户${userId}创建了${count}个档案`);

// 改造后
log.info('用户创建档案', { userId, count });
```
