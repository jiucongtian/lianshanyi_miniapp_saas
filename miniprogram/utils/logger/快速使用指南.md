# Logger 快速使用指南

## 核心概念

`createModuleLogger` 返回的方法签名与 `BaseClass._log` 保持一致：

```javascript
log.info(methodName, message, data?)
```

**输出格式：**
```
[2025-10-12 17:47:48.340] [INFO] [模块名] [模块名:方法名] 日志信息
```

## 基本使用

### 1. 工具函数中使用

```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('UserManager');

function getUserInfo(userId) {
  log.info('getUserInfo', '开始获取用户信息', { userId });
  
  try {
    // 业务逻辑
    log.info('getUserInfo', '获取成功', { userId });
  } catch (error) {
    log.error('getUserInfo', '获取失败', { userId, error: error.message });
  }
}
```

### 2. 类中使用

```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('ImageCache');

class ImageCacheManager {
  constructor() {
    log.info('constructor', '初始化缓存管理器');
  }
  
  async cacheImage(url) {
    log.info('cacheImage', '开始缓存图片', { url });
    
    try {
      // 业务逻辑
      log.info('cacheImage', '缓存成功', { url });
    } catch (error) {
      log.error('cacheImage', '缓存失败', { url, error: error.message });
    }
  }
}
```

### 3. 页面中使用

```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('ProfilePage');

Page({
  onLoad(options) {
    log.info('onLoad', '页面加载', { options });
  },
  
  onButtonTap(e) {
    const id = e.currentTarget.dataset.id;
    log.info('onButtonTap', '按钮点击', { id });
  },
  
  onShareAppMessage() {
    log.info('onShareAppMessage', '分享小程序');
    return {
      title: '分享标题',
      path: '/pages/index/index'
    };
  }
});
```

### 4. 对象方法中使用

```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('ProfileManager');

const profileManager = {
  async loadProfile(profileId) {
    log.info('loadProfile', '加载档案', { profileId });
    
    try {
      const data = await this.fetchData(profileId);
      log.info('loadProfile', '加载成功', { profileId });
      return data;
    } catch (error) {
      log.error('loadProfile', '加载失败', { profileId, error: error.message });
      throw error;
    }
  },
  
  deleteProfile(profileId) {
    log.warn('deleteProfile', '删除档案', { profileId });
    // 业务逻辑
  }
};
```

## 日志级别选择

| 级别 | 使用场景 | 是否输出到生产环境 |
|-----|---------|------------------|
| `debug` | 调试信息、内部状态 | ❌ 否 |
| `info` | 关键流程、成功操作 | ❌ 否（仅存储） |
| `warn` | 警告、降级处理 | ✅ 是 |
| `error` | 错误、异常 | ✅ 是 |

## 性能追踪

```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('BaziService');

async function calculateBazi(birthInfo) {
  log.info('calculateBazi', '开始计算八字', { birthInfo });
  
  // 开始性能追踪
  log.timeStart('calculate-bazi');
  
  try {
    const result = await performCalculation(birthInfo);
    
    // 结束性能追踪（会自动输出耗时）
    log.timeEnd('calculate-bazi');
    
    log.info('calculateBazi', '计算完成', { result });
    return result;
  } catch (error) {
    log.error('calculateBazi', '计算失败', { birthInfo, error: error.message });
    throw error;
  }
}
```

## 常见模式

### 异步函数模式

```javascript
async function updateData(id, data) {
  log.info('updateData', '开始更新', { id });
  
  try {

    log.error('updateData', '更新失败', { id, error: error.message });
    throw error;
  }

```

```javascript
  }
  
  const hasPermission = // 权限检查逻辑

    log.warn('checkPermission', '权限不足', { userId, permission });
```

### 循环操作模式

```javascript
async function batchProcess(items) {
  log.info('batchProcess', '开始批量处理', { count: items.length });
  
  const results = [];
  let failCount = 0;
  
  for (const item of items) {
    try {
      const result = await processItem(item);
      results.push(result);
    } catch (error) {
      failCount++;
      log.error('batchProcess', '处理失败', { 
        itemId: item.id, 
        error: error.message 
      });
    }
  }
  
  log.info('batchProcess', '批量处理完成', { 
    total: items.length,
    success: results.length,
    failed: failCount
  });
  
  return results;
}
```

## 最佳实践

### ✅ 推荐

```javascript
// 1. 结构化数据
log.info('login', '用户登录', { userId, timestamp: Date.now() });

// 2. 方法名清晰
log.info('getUserProfile', '获取用户档案', { profileId });

// 3. 错误信息详细
log.error('saveData', '保存失败', { 
  id, 
  error: error.message,
  code: error.code 
});

// 4. 性能追踪耗时操作
log.timeStart('heavy-task');
// 耗时操作
log.timeEnd('heavy-task');
```

### ❌ 不推荐

```javascript
// 1. 字符串拼接
log.info('login', `用户${userId}在${time}登录`);

// 2. 方法名不明确
log.info('do', '操作');

// 3. 错误信息不足
log.error('error', '失败了');

// 4. 敏感信息未脱敏
log.info('login', '登录', { 
  password: '123456',  // ❌ 不要记录密码
  phone: '13800138000' // ❌ 应该脱敏
});
```

## 敏感信息处理

Logger 会自动过滤以下字段：
- `password`
- `token`
- `openid`
- `sessionKey`
- `phoneNumber`

如果需要记录其他敏感信息，请手动脱敏：

```javascript
// 手机号脱敏
const safePhone = phone.substring(0, 3) + '****' + phone.substring(7);
log.info('register', '用户注册', { phone: safePhone });

// 身份证脱敏
const safeIdCard = idCard.substring(0, 6) + '********' + idCard.substring(14);
log.info('verify', '实名认证', { idCard: safeIdCard });
```

## 快速模板

### 文件头部模板

```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('模块名');
```

### 函数模板

```javascript
function functionName(params) {
  log.info('functionName', '描述', { params });
  
  try {
    // 业务逻辑
    log.info('functionName', '成功');
  } catch (error) {
    log.error('functionName', '失败', { error: error.message });
    throw error;
  }
}
```

### 异步函数模板

```javascript
async function asyncFunction(params) {
  log.info('asyncFunction', '开始', { params });
  
  log.timeStart('async-operation');
  try {
    const result = await someAsyncOperation();
    log.timeEnd('async-operation');
    log.info('asyncFunction', '成功', { result });
    return result;
  } catch (error) {
    log.error('asyncFunction', '失败', { error: error.message });
    throw error;
  }
}
```

## 对比：BaseClass 与 createModuleLogger

### BaseClass（继承类使用）

```javascript
class ProfileController extends BaseController {
  async loadData() {
    // this._log 会自动添加类名
    this._log('loadData', '加载数据');
    // 输出：[ProfileController] [ProfileController:loadData] 加载数据
  }
}
```

### createModuleLogger（非继承类使用）

```javascript
const { createModuleLogger } = require('@logger/');
const log = createModuleLogger('ProfileManager');

function loadData() {
  // 手动指定方法名
  log.info('loadData', '加载数据');
  // 输出：[ProfileManager] [ProfileManager:loadData] 加载数据
}
```

**两种方式输出格式完全一致！** 🎉
