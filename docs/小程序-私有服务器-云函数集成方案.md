# 小程序-私有服务器-云函数集成方案

## 🏗️ 架构概述

本方案实现了一个三层架构：
```
微信小程序 → 私有服务器 → 腾讯云函数
    ↓           ↓           ↓
 用户交互    业务逻辑处理   数据计算处理
```

### 架构优势
- 🔒 **安全性**：私有服务器作为中间层，保护敏感信息
- 🚀 **性能**：服务器端缓存和批量处理
- 🔄 **灵活性**：可以组合多个云函数调用
- 📊 **监控**：统一的日志和监控

#### 1.2 小程序API封装

创建 `miniprogram/api/privateServer.js`：

```javascript
/**
 * 私有服务器API调用封装
 */

const CONFIG = {
  baseURL: 'https://your-private-server.com/api',
  timeout: 10000
}

/**
 * 通用请求方法
 * @param {string} url - 接口路径
 * @param {object} data - 请求数据
 * @param {string} method - 请求方法
 * @returns {Promise}
 */
function request(url, data = {}, method = 'POST') {
  return new Promise((resolve, reject) => {
    // 获取用户登录态
    wx.checkSession({
      success: () => {
        // 登录态有效，直接调用
        makeRequest();
      },
      fail: () => {
        // 登录态过期，重新登录
        wx.login({
          success: (loginRes) => {
            data.code = loginRes.code;
            makeRequest();
          },
          fail: reject
        });
      }
    });

    function makeRequest() {
      wx.request({
        url: `${CONFIG.baseURL}${url}`,
        data: {
          ...data,
          appid: wx.getAccountInfoSync().miniProgram.appId,
          timestamp: Date.now()
        },
        method,
        header: {
          'Content-Type': 'application/json',
          'User-Agent': 'MiniProgram'
        },
        timeout: CONFIG.timeout,
        success: (res) => {
          if (res.statusCode === 200) {
            resolve(res.data);
          } else {
            reject(new Error(`HTTP ${res.statusCode}: ${res.data?.message || 'Unknown error'}`));
          }
        },
        fail: (error) => {
          reject(new Error(`Network error: ${error.errMsg}`));
        }
      });
    }
  });
}

/**
 * 调用生辰八字计算
 * @param {number} timestamp - 时间戳
 * @returns {Promise}
 */
export function calculateBazi(timestamp) {
  return request('/bazi/calculate', {
    timestamp,
    source: 'miniprogram'
  });
}

/**
 * 调用测试函数
 * @param {object} params - 测试参数
 * @returns {Promise}
 */
export function testFunction(params) {
  return request('/test/function', {
    ...params,
    source: 'miniprogram'
  });
}

/**
 * 批量调用云函数
 * @param {Array} functions - 函数列表
 * @returns {Promise}
 */
export function batchCallFunctions(functions) {
  return request('/batch/call', {
    functions,
    source: 'miniprogram'
  });
}
```

#### 2.4 微信服务封装 (src/services/wechatService.js)

```javascript
const axios = require('axios');
const config = require('../config');

class WechatService {
  constructor() {
    this.accessToken = null;
    this.tokenExpireTime = 0;
  }

  /**
   * 获取access_token
   * @returns {Promise<string>}
   */
  async getAccessToken() {
    const now = Date.now();
    
    // 如果token还有效，直接返回
    if (this.accessToken && now < this.tokenExpireTime) {
      return this.accessToken;
    }

    try {
      const response = await axios.get('https://api.weixin.qq.com/cgi-bin/token', {
        params: {
          grant_type: 'client_credential',
          appid: config.wechat.appid,
          secret: config.wechat.secret
        },
        timeout: 10000
      });

      if (response.data.access_token) {
        this.accessToken = response.data.access_token;
        // 提前5分钟过期，确保安全
        this.tokenExpireTime = now + (response.data.expires_in - 300) * 1000;
        
        console.log('Access token refreshed successfully');
        return this.accessToken;
      } else {
        throw new Error(`Failed to get access token: ${response.data.errmsg}`);
      }
    } catch (error) {
      console.error('Get access token failed:', error);
      throw new Error('Failed to authenticate with WeChat API');
    }
  }

  /**
   * 验证用户登录态
   * @param {string} code - 小程序登录凭证
   * @returns {Promise<object>}
   */
  async verifyLogin(code) {
    if (!code) {
      throw new Error('Login code is required');
    }

    try {
      const response = await axios.get('https://api.weixin.qq.com/sns/jscode2session', {
        params: {
          appid: config.wechat.appid,
          secret: config.wechat.secret,
          js_code: code,
          grant_type: 'authorization_code'
        }
      });

      if (response.data.errcode) {
        throw new Error(`WeChat login failed: ${response.data.errmsg}`);
      }

      return {
        openid: response.data.openid,
        sessionKey: response.data.session_key,
        unionid: response.data.unionid
      };
    } catch (error) {
      console.error('Verify login failed:', error);
      throw new Error('Failed to verify user login');
    }
  }
}

module.exports = new WechatService();
```

#### 2.5 云函数服务封装 (src/services/cloudService.js)

```javascript
const axios = require('axios');
const wechatService = require('./wechatService');
const config = require('../config');

class CloudService {
  /**
   * 调用腾讯云函数
   * @param {string} functionName - 云函数名称
   * @param {object} data - 传递给云函数的数据
   * @returns {Promise<object>}
   */
  async invokeCloudFunction(functionName, data = {}) {
    try {
      // 获取access_token
      const accessToken = await wechatService.getAccessToken();

      // 构建请求URL
      const url = 'https://api.weixin.qq.com/tcb/invokecloudfunction';
      const params = {
        access_token: accessToken,
        env: config.tencent.cloudEnv,
        name: functionName
      };

      console.log(`Invoking cloud function: ${functionName}`);
      console.log('Request data:', data);

      // 调用云函数
      const response = await axios.post(url, data, {
        params,
        headers: {
          'Content-Type': 'application/json'
        },
        timeout: 30000
      });

      // 检查响应
      if (response.data.errcode === 0) {
        const result = JSON.parse(response.data.resp_data);
        console.log(`Cloud function ${functionName} executed successfully`);
        return {
          success: true,
          data: result,
          functionName,
          executionTime: new Date().toISOString()
        };
      } else {
        throw new Error(`Cloud function error: ${response.data.errmsg}`);
      }
    } catch (error) {
      console.error(`Cloud function ${functionName} failed:`, error);
      
      // 处理不同类型的错误
      if (error.response) {
        throw new Error(`HTTP ${error.response.status}: ${error.response.data?.errmsg || error.response.statusText}`);
      } else if (error.request) {
        throw new Error('Network request failed');
      } else {
        throw new Error(error.message || 'Unknown error occurred');
      }
    }
  }

  /**
   * 批量调用云函数
   * @param {Array} functions - 函数配置数组 [{name, data}, ...]
   * @returns {Promise<Array>}
   */
  async batchInvokeCloudFunctions(functions) {
    const results = [];
    const errors = [];

    // 并行调用所有函数
    const promises = functions.map(async (func, index) => {
      try {
        const result = await this.invokeCloudFunction(func.name, func.data);
        results[index] = result;
      } catch (error) {
        errors[index] = {
          functionName: func.name,
          error: error.message
        };
      }
    });

    await Promise.allSettled(promises);

    return {
      success: errors.length === 0,
      results,
      errors: errors.filter(Boolean),
      totalCount: functions.length,
      successCount: results.filter(Boolean).length,
      errorCount: errors.filter(Boolean).length
    };
  }

  /**
   * 调用生辰八字计算云函数
   * @param {number} timestamp - 时间戳
   * @returns {Promise<object>}
   */
  async calculateBazi(timestamp) {
    return this.invokeCloudFunction('calculateBazi', { timestamp });
  }

  /**
   * 调用测试云函数
   * @param {object} params - 测试参数
   * @returns {Promise<object>}
   */
  async testFunction(params) {
    return this.invokeCloudFunction('testFunction', params);
  }
}

module.exports = new CloudService();
```

#### 2.6 路由实现

**生辰八字路由 (src/routes/bazi.js)：**

```javascript
const express = require('express');
const cloudService = require('../services/cloudService');
const { validateTimestamp } = require('../middleware/validate');

const router = express.Router();

/**
 * POST /api/bazi/calculate
 * 计算生辰八字
 */
router.post('/calculate', validateTimestamp, async (req, res) => {
  try {
    const { timestamp, source } = req.body;
    
    console.log(`Bazi calculation request from ${source}:`, { timestamp });

    // 调用云函数
    const result = await cloudService.calculateBazi(timestamp);

    if (result.success) {
      res.json({
        success: true,
        message: '计算成功',
        data: result.data,
        meta: {
          timestamp: result.executionTime,
          source,
          functionName: result.functionName
        }
      });
    } else {
      res.status(500).json({
        success: false,
        message: '计算失败',
        error: result.error
      });
    }
  } catch (error) {
    console.error('Bazi calculation error:', error);
    res.status(500).json({
      success: false,
      message: '服务器内部错误',
      error: error.message
    });
  }
});

module.exports = router;
```

**批量处理路由 (src/routes/batch.js)：**

```javascript
const express = require('express');
const cloudService = require('../services/cloudService');

const router = express.Router();

/**
 * POST /api/batch/call
 * 批量调用云函数
 */
router.post('/call', async (req, res) => {
  try {
    const { functions, source } = req.body;
    
    if (!Array.isArray(functions) || functions.length === 0) {
      return res.status(400).json({
        success: false,
        message: '函数列表不能为空'
      });
    }

    console.log(`Batch call request from ${source}:`, functions.map(f => f.name));

    // 批量调用云函数
    const result = await cloudService.batchInvokeCloudFunctions(functions);

    res.json({
      success: result.success,
      message: `批量调用完成，成功 ${result.successCount}/${result.totalCount}`,
      data: {
        results: result.results,
        summary: {
          totalCount: result.totalCount,
          successCount: result.successCount,
          errorCount: result.errorCount
        }
      },
      errors: result.errors
    });
  } catch (error) {
    console.error('Batch call error:', error);
    res.status(500).json({
      success: false,
      message: '批量调用失败',
      error: error.message
    });
  }
});

module.exports = router;
```

#### 2.7 主应用文件 (src/app.js)

```javascript
const express = require('express');
const cors = require('cors');
const helmet = require('helmet');
require('dotenv').config();

const baziRoutes = require('./routes/bazi');
const testRoutes = require('./routes/test');
const batchRoutes = require('./routes/batch');

const app = express();
const PORT = process.env.PORT || 3000;

// 中间件
app.use(helmet());
app.use(cors({
  origin: ['https://servicewechat.com'], // 微信小程序域名
  credentials: true
}));
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: true }));

// 请求日志
app.use((req, res, next) => {
  console.log(`${new Date().toISOString()} ${req.method} ${req.path}`);
  next();
});

// 路由
app.use('/api/bazi', baziRoutes);
app.use('/api/test', testRoutes);
app.use('/api/batch', batchRoutes);

// 健康检查
app.get('/health', (req, res) => {
  res.json({
    status: 'healthy',
    timestamp: new Date().toISOString(),
    version: '1.0.0'
  });
});

// 错误处理
app.use((error, req, res, next) => {
  console.error('Unhandled error:', error);
  res.status(500).json({
    success: false,
    message: '服务器内部错误',
    error: process.env.NODE_ENV === 'development' ? error.message : 'Internal server error'
  });
});

// 404处理
app.use('*', (req, res) => {
  res.status(404).json({
    success: false,
    message: '接口不存在'
  });
});

app.listen(PORT, () => {
  console.log(`Private server is running on port ${PORT}`);
  console.log(`Environment: ${process.env.NODE_ENV}`);
});

module.exports = app;
```
